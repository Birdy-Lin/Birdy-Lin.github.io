// 角色护甲系统（类似RPG血条）
class CharacterArmor {
constructor(persona) {
this.outer = { hp: 100, layer: "傲娇嘴硬" }; // 外层护甲
this.middle = { hp: 200, layer: "动摇辩解" }; // 中层护甲
this.inner = { hp: 300, layer: "内心脆弱" }; // 内层护甲
}
}
// 攻防判定逻辑（每回合）
function evaluateBattle(playerPrompt, character) {
const aiResponse = await callLLM(
systemPrompt: character.armor, // 角色System Prompt
userInput: playerPrompt // 玩家输入
);
const damage = analyzeSentiment(aiResponse); // 解析情绪值
armor.hp -= damage; // 扣除护甲
affection += calculateAffection(damage); // 计算好感度
return { isBreached: armor.hp <= 0 }; // 护甲是否被击破
}
